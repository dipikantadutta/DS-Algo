public class StackAssignment
{
    static class Stack
    {
        static int top = -1;
        static ArrayList <Student> list = new ArrayList<Student>();
        static void push(Student student)
        {
            top++;
            list.add(student);

        }
        
        static Student pop()
        {
            if (top == -1)
                return null;
            else
            {
                Student st = list.get(list.size()-1);
                top--;
                return st;
            }
        }
        
        static Student peek()
        {
            if (top == -1)
                return null;
            else
            {
                Student st = list.get(list.size()-1);
                return st;
            }
        }
        
        boolean isEmpty()
        {
            return ((top==-1)?true:false);
        }
        
    
    }
